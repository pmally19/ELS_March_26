### ✅ Table 2: `posting_period_controls`
**Purpose:** Fine-grained control over posting permissions
**Status:** ✅ FULLY IMPLEMENTED
**Location:** `database/migrations/create-posting-period-controls-table.sql`

```sql
CREATE TABLE posting_period_controls (
  id SERIAL PRIMARY KEY,
  
  -- Company and Fiscal Year
  company_code_id INTEGER NOT NULL REFERENCES company_codes(id),
  fiscal_year_variant_id INTEGER REFERENCES fiscal_year_variants(id),
  fiscal_year INTEGER NOT NULL,
  
  -- Period Range
  period_from INTEGER NOT NULL CHECK (period_from >= 1 AND period_from <= 16),
  period_to INTEGER NOT NULL CHECK (period_to >= 1 AND period_to <= 16),
  
  -- Control Flags
  posting_status VARCHAR(20) NOT NULL DEFAULT 'OPEN' 
    CHECK (posting_status IN ('OPEN', 'CLOSED', 'LOCKED')),
  allow_posting BOOLEAN NOT NULL DEFAULT TRUE,
  allow_adjustments BOOLEAN NOT NULL DEFAULT FALSE,
  allow_reversals BOOLEAN NOT NULL DEFAULT TRUE,
  
  -- Audit Trail
  control_reason TEXT,
  controlled_by INTEGER,
  controlled_at TIMESTAMP,
  is_active BOOLEAN NOT NULL DEFAULT TRUE,
  
  -- Timestamps
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW(),
  created_by INTEGER,
  updated_by INTEGER,
  
  CONSTRAINT check_period_range CHECK (period_to >= period_from),
  CONSTRAINT unique_company_fiscal_period 
    UNIQUE (company_code_id, fiscal_year, period_from, period_to)
);
```

**Key Features:**
1. **Period Ranges:** Can control periods 1-3 separately from 4-12
2. **Three Status Levels:**
   - `OPEN`: Full posting allowed
   - `CLOSED`: Limited posting (adjustments if flag set)
   - `LOCKED`: No posting whatsoever
3. **Granular Permissions:**
   - `allow_posting`: Regular transactions
   - `allow_adjustments`: Correction entries
   - `allow_reversals`: Reversal documents
4. **Audit Trail:** Who closed/locked and when

**Real-World Example:**
```sql
-- Close periods 1-11 of 2024 for company 1000
INSERT INTO posting_period_controls 
(company_code_id, fiscal_year, period_from, period_to, posting_status)
VALUES (1, 2024, 1, 11, 'CLOSED');

-- Keep period 12 open for year-end adjustments
INSERT INTO posting_period_controls 
(company_code_id, fiscal_year, period_from, period_to, 
 posting_status, allow_adjustments)
VALUES (1, 2024, 12, 12, 'CLOSED', TRUE);
```

**1.1 Add Period Lock Validation to All Posting Endpoints**

```typescript
export async function checkPeriodLock(
  companyCodeId: number,
  postingDate: Date,
  documentType: string = 'normal'
) {
  const period = postingDate.getMonth() + 1;
  const fiscalYear = postingDate.getFullYear();
  
  const result = await pool.query(`
    SELECT posting_status, allow_posting, allow_adjustments, allow_reversals, control_reason
    FROM posting_period_controls
    WHERE company_code_id = $1
      AND fiscal_year = $2
      AND period_from <= $3
      AND period_to >= $3
      AND is_active = TRUE
    LIMIT 1
  `, [companyCodeId, fiscalYear, period]);
  
  if (result.rows.length > 0) {
    const control = result.rows[0];
    
    if (control.posting_status === 'LOCKED') {
      throw new Error(`Period ${period}/${fiscalYear} is LOCKED. ${control.control_reason || ''}`);
    }
    
    if (!control.allow_posting && documentType === 'normal') {
      throw new Error(`Period ${period}/${fiscalYear} is CLOSED. Only adjustments allowed.`);
    }
    
    if (documentType === 'adjustment' && !control.allow_adjustments) {
      throw new Error(`Adjustments not allowed for period ${period}/${fiscalYear}.`);
    }
  }
  
  return true;
}
```

**Use in all posting routes:**
```typescript
await checkPeriodLock(companyCodeId, postingDate);
```

```typescript
// In server/routes/accounting-documents.ts or gl-entry-routes.ts

router.post('/accounting-documents', async (req, res) => {
  try {
    const { posting_date, company_code_id, ...rest } = req.body;
    
    // ⭐ ADD THIS CHECK
    const postingDate = new Date(posting_date);
    const period = postingDate.getMonth() + 1;
    const fiscalYear = postingDate.getFullYear();
    
    const periodCheck = await pool.query(`
      SELECT * FROM posting_period_controls
      WHERE company_code_id = $1
        AND fiscal_year = $2
        AND period_from <= $3
        AND period_to >= $3
        AND is_active = TRUE
      LIMIT 1
    `, [company_code_id, fiscalYear, period]);
    
    if (periodCheck.rows.length > 0) {
      const control = periodCheck.rows[0];
      if (control.posting_status === 'LOCKED' || !control.allow_posting) {
        return res.status(403).json({
          error: `Period ${period}/${fiscal Year} is ${control.posting_status}`,
          reason: control.control_reason
        });
      }
    }
    
    // Continue with posting...
  } catch (error) {
    // Handle error
  }
});
```

### Sample 2: Frontend Component Skeleton

```typescript
// client/src/pages/finance/PeriodEndClosing.tsx

import { useQuery, useMutation } from '@tanstack/react-query';

export function PeriodEndClosing() {
  const { data: periods } = useQuery({
    queryKey: ['fiscal-periods'],
    queryFn: () => fetch('/api/fiscal-periods').then(r => r.json())
  });
  
  const closePeriodMutation = useMutation({
    mutationFn: async (periodId: number) => {
      // Create closing record
      const closing = await fetch('/api/period-end-closing', {
        method: 'POST',
        body: JSON.stringify({ fiscalPeriodId: periodId })
      }).then(r => r.json());
      
      // Process closing
      return fetch(`/api/period-end-closing/${closing.id}/process`, {
        method: 'POST'
      }).then(r => r.json());
    },
    onSuccess: () => {
      // Refetch periods
    }
  });
  
  return (
    <div>
      <h1>Period End Closing</h1>
      <table>
        {periods?.map(period => (
          <tr key={period.id}>
            <td>{period.year} - {period.period}</td>
            <td>{period.status}</td>
            <td>
              {period.posting_allowed ? (
                <button onClick={() => closePeriodMutation.mutate(period.id)}>
                  Close Period
                </button>
              ) : (
                <span>Closed</span>
              )}
            </td>
          </tr>
        ))}
      </table>
    </div>
  );
}
```
